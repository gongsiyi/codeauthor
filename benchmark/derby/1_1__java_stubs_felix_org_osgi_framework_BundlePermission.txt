1:a8b6921: 						&& (a[i - 4] == 'q' || a[i - 4] == 'Q')
1:a8b6921: 						&& (a[i - 3] == 'u' || a[i - 3] == 'U')
1:a8b6921: 						&& (a[i - 2] == 'i' || a[i - 2] == 'I')

1:a8b6921: 						&& (a[i] == 'e' || a[i] == 'E')) {
1:a8b6921: 					matchlen = 7;
