1:         ResultSetMetaData rsmd = rs.getMetaData();

1:         if (numCols <= 0) return;
1: 
1:         StringBuffer heading = new StringBuffer();	
1:         for (int i=1; i<=numCols; i++)
1:         {
1:             if (i > 1) heading.append(",");
1:             heading.append("COL"+i);
1:             heading.append("(datatype : " + rsmd.getColumnTypeName(i));
1:             heading.append(", precision : " + rsmd.getPrecision(i));
1:             heading.append(", scale : " + rsmd.getScale(i) + ")");
1: 
1:         }
1: 
1:         StringBuffer row = new StringBuffer();

1:         {
1:             for (int i=1; i<=numCols; i++)
1:             {
1:                 if (i > 1) row.append(",");
1:                 row.append(rs.getString(i));
1:                 row.append(" ");	
1:             }
1: 
1:         }
1: 
1:         String actualValue = heading.toString() + " " + row.toString();
1: 
1:         if ( usingDerbyNetClient() ) {

1:                 String actualValue2  = actualValue.replaceAll("CHAR", "CHAR ()");

1:             } 
1: 
1:         }
1:         assertEquals(expectedValue, actualValue);	
1:         rs.close();
1: 
1:     }
